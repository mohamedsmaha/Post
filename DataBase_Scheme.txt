
User {
	id integer pk increments unique
	Email text unique
	PhoneNumber integer unique
	UserName text
	Password text
	BirthDay integer
	Img text > Imges.id
	Lang integer > Languagh.id
	ColorTheme integer
	Api_Token text
	Secound_Api_token text
	session integer
}

	id integer pk increments
	User integer *>* User.id
	Friend integer *>* User.id
}

	id integer pk increments
	Text text
}

Imges {

}

Sponser {
	id integer pk increments
